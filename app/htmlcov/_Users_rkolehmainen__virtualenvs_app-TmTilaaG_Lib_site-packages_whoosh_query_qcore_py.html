60def token_lists(q, phrases=True): 

61 """Returns the terms in the query tree, with the query hierarchy 

62 represented as nested lists. 

63 """ 

64 

65 if q.is_leaf(): 

66 from whoosh.query import Phrase 

67 if phrases or not isinstance(q, Phrase): 

68 return list(q.tokens()) 

69 else: 

70 ls = [] 

71 for qq in q.children(): 

72 t = token_lists(qq, phrases=phrases) 

73 if len(t) == 1: 

74 t = t[0] 

75 if t: 

76 ls.append(t) 

77 return ls

239 def apply(self, fn): 

240 """If this query has children, calls the given function on each child 

241 and returns a new copy of this node with the new children returned by 

242 the function. If this is a leaf node, simply returns this object. 

243 

244 This is useful for writing functions that transform a query tree. For 

245 example, this function changes all Term objects in a query tree into 

246 Variations objects:: 

247 

248 def term2var(q): 

249 if isinstance(q, Term): 

250 return Variations(q.fieldname, q.text) 

251 else: 

252 return q.apply(term2var)

254 q = And([Term("f", "alfa"), 

255 Or([Term("f", "bravo"), 

256 Not(Term("f", "charlie"))])]) 

257 q = term2var(q) 

258 

259 Note that this method does not automatically create copies of nodes. 

260 To avoid modifying the original tree, your function should call the 

261 :meth:`Query.copy` method on nodes before changing their attributes. 

262 """

338 def existing_terms(self, ixreader, phrases=True, expand=False, fieldname=None): 

339 """Returns a set of all byteterms in this query tree that exist in 

340 the given ixreader. 

341 

342 :param ixreader: A :class:`whoosh.reading.IndexReader` object. 

343 :param phrases: Whether to add words found in Phrase queries. 

344 :param expand: If True, queries that match multiple terms 

345 will return all matching expansions. 

346 :rtype: set 

347 """ 

348 

349 schema = ixreader.schema 

350 termset = set() 

351 

352 for q in self.leaves(): 

353 if fieldname and fieldname != q.field(): 

354 continue 

355 

356 if expand: 

357 terms = q.expanded_terms(ixreader, phrases=phrases) 

358 else: 

359 terms = q.terms(phrases=phrases) 

360 

361 for fieldname, text in terms: 

362 if (fieldname, text) in termset: 

363 continue 

364 

365 if fieldname in schema: 

366 field = schema[fieldname] 

367 

368 try: 

369 btext = field.to_bytes(text) 

370 except ValueError: 

371 continue 

372 

373 if (fieldname, btext) in ixreader: 

374 termset.add((fieldname, btext)) 

375 return termset

627class Every(Query): 

628 """A query that matches every document containing any term in a given 

629 field. If you don't specify a field, the query matches every document. 

630 

631 >>> # Match any documents with something in the "path" field 

632 >>> q = Every("path") 

633 >>> # Matcher every document 

634 >>> q = Every() 

635 

636 The unfielded form (matching every document) is efficient. 

637 

638 The fielded is more efficient than a prefix query with an empty prefix or a 

639 '*' wildcard, but it can still be very slow on large indexes. It requires 

640 the searcher to read the full posting list of every term in the given 

641 field. 

642 

643 Instead of using this query it is much more efficient when you create the 

644 index to include a single term that appears in all documents that have the 

645 field you want to match. 

646 

647 For example, instead of this:: 

648 

649 # Match all documents that have something in the "path" field 

650 q = Every("path") 

651 

652 Do this when indexing:: 

653 

654 # Add an extra field that indicates whether a document has a path 

655 schema = fields.Schema(path=fields.ID, has_path=fields.ID) 

656 

657 # When indexing, set the "has_path" field based on whether the document 

658 # has anything in the "path" field 

659 writer.add_document(text=text_value1) 

660 writer.add_document(text=text_value2, path=path_value2, has_path="t") 

661 

662 Then to find all documents with a path:: 

663 

664 q = Term("has_path", "t") 

665 """ 

666 

667 def __init__(self, fieldname=None, boost=1.0): 

668 """ 

669 :param fieldname: the name of the field to match, or ``None`` or ``*`` 

670 to match all documents. 

671 """ 

672 

673 if not fieldname or fieldname == "*": 

674 fieldname = None 

675 self.fieldname = fieldname 

676 self.boost = boost 

677 

678 def __repr__(self): 

679 return "%s(%r, boost=%s)" % (self.__class__.__name__, self.fieldname, 

680 self.boost) 

681 

682 def __eq__(self, other): 

683 return (other and self.__class__ is other.__class__ 

684 and self.fieldname == other.fieldname 

685 and self.boost == other.boost) 

686 

687 def __unicode__(self): 

688 return u("%s:*") % self.fieldname 

689 

690 __str__ = __unicode__ 

691 

692 def __hash__(self): 

693 return hash(self.fieldname)

695 def estimate_size(self, ixreader): 

696 return ixreader.doc_count()

698 def matcher(self, searcher, context=None): 

699 fieldname = self.fieldname 

700 reader = searcher.reader() 

701 

702 if fieldname in (None, "", "*"): 

703 # This takes into account deletions 

704 doclist = array("I", reader.all_doc_ids()) 

705 else: 

706 # This is a hacky hack, but just create an in-memory set of all the 

707 # document numbers of every term in the field. This is SLOOOW for 

708 # large indexes 

709 doclist = set() 

710 for text in searcher.lexicon(fieldname): 

711 pr = searcher.postings(fieldname, text) 

712 doclist.update(pr.all_ids()) 

713 doclist = sorted(doclist) 

714 

715 return matching.ListMatcher(doclist, all_weights=self.boost)
