35class StemFilter(Filter): 

36 """Stems (removes suffixes from) the text of tokens using the Porter 

37 stemming algorithm. Stemming attempts to reduce multiple forms of the same 

38 root word (for example, "rendering", "renders", "rendered", etc.) to a 

39 single word in the index. 

40 

41 >>> stemmer = RegexTokenizer() | StemFilter() 

42 >>> [token.text for token in stemmer("fundamentally willows")] 

43 ["fundament", "willow"] 

44 

45 You can pass your own stemming function to the StemFilter. The default 

46 is the Porter stemming algorithm for English. 

47 

48 >>> stemfilter = StemFilter(stem_function) 

49 

50 You can also use one of the Snowball stemming functions by passing the 

51 `lang` keyword argument. 

52 

53 >>> stemfilter = StemFilter(lang="ru") 

54 

55 The list of available languages is in `whoosh.lang.languages`. 

56 You can use :func:`whoosh.lang.has_stemmer` to check if a given language has 

57 a stemming function available. 

58 

59 By default, this class wraps an LRU cache around the stemming function. The 

60 ``cachesize`` keyword argument sets the size of the cache. To make the 

61 cache unbounded (the class caches every input), use ``cachesize=-1``. To 

62 disable caching, use ``cachesize=None``. 

63 

64 If you compile and install the py-stemmer library, the 

65 :class:`PyStemmerFilter` provides slightly easier access to the language 

66 stemmers in that library. 

67 """ 

68 

69 __inittypes__ = dict(stemfn=object, ignore=list) 

70 

71 is_morph = True

73 def __init__(self, stemfn=stem, lang=None, ignore=None, cachesize=50000): 

74 """ 

75 :param stemfn: the function to use for stemming. 

76 :param lang: if not None, overrides the stemfn with a language stemmer 

77 from the ``whoosh.lang.snowball`` package. 

78 :param ignore: a set/list of words that should not be stemmed. This is 

79 converted into a frozenset. If you omit this argument, all tokens 

80 are stemmed. 

81 :param cachesize: the maximum number of words to cache. Use ``-1`` for 

82 an unbounded cache, or ``None`` for no caching. 

83 """ 

84 

85 self.stemfn = stemfn 

86 self.lang = lang 

87 self.ignore = frozenset() if ignore is None else frozenset(ignore) 

88 self.cachesize = cachesize 

89 # clear() sets the _stem attr to a cached wrapper around self.stemfn 

90 self.clear()

92 def __getstate__(self): 

93 # Can't pickle a dynamic function, so we have to remove the _stem 

94 # attribute from the state 

95 return dict([(k, self.__dict__[k]) for k in self.__dict__ 

96 if k != "_stem"])

98 def __setstate__(self, state): 

99 # Check for old instances of StemFilter class, which didn't have a 

100 # cachesize attribute and pickled the cache attribute 

101 if "cachesize" not in state: 

102 self.cachesize = 50000 

103 if "ignores" in state: 

104 self.ignore = state["ignores"] 

105 elif "ignore" not in state: 

106 self.ignore = frozenset() 

107 if "lang" not in state: 

108 self.lang = None 

109 if "cache" in state: 

110 del state["cache"] 

111 

112 self.__dict__.update(state) 

113 # Set the _stem attribute 

114 self.clear()

116 def clear(self): 

117 if self.lang: 

118 from whoosh.lang import stemmer_for_language 

119 stemfn = stemmer_for_language(self.lang) 

120 else: 

121 stemfn = self.stemfn 

122 

123 if isinstance(self.cachesize, integer_types) and self.cachesize != 0: 

124 if self.cachesize < 0: 

125 self._stem = unbound_cache(stemfn) 

126 elif self.cachesize > 1: 

127 self._stem = lfu_cache(self.cachesize)(stemfn) 

128 else: 

129 self._stem = stemfn 

130 

131 def cache_info(self): 

132 if self.cachesize <= 1: 

133 return None 

134 return self._stem.cache_info() 

135 

136 def __eq__(self, other): 

137 return (other and self.__class__ is other.__class__ 

138 and self.stemfn == other.stemfn) 

139 

140 def __call__(self, tokens): 

141 stemfn = self._stem 

142 ignore = self.ignore 

143 

144 for t in tokens: 

145 if not t.stopped: 

146 text = t.text 

147 if text not in ignore: 

148 t.text = stemfn(text) 

149 yield t

219class DoubleMetaphoneFilter(Filter): 

220 """Transforms the text of the tokens using Lawrence Philips's Double 

221 Metaphone algorithm. This algorithm attempts to encode words in such a way 

222 that similar-sounding words reduce to the same code. This may be useful for 

223 fields containing the names of people and places, and other uses where 

224 tolerance of spelling differences is desireable. 

225 """ 

226 

227 is_morph = True 

228 

229 def __init__(self, primary_boost=1.0, secondary_boost=0.5, combine=False): 

230 """ 

231 :param primary_boost: the boost to apply to the token containing the 

232 primary code. 

233 :param secondary_boost: the boost to apply to the token containing the 

234 secondary code, if any. 

235 :param combine: if True, the original unencoded tokens are kept in the 

236 stream, preceding the encoded tokens. 

237 """ 

238 

239 self.primary_boost = primary_boost 

240 self.secondary_boost = secondary_boost 

241 self.combine = combine 

242 

243 def __eq__(self, other): 

244 return (other 

245 and self.__class__ is other.__class__ 

246 and self.primary_boost == other.primary_boost) 

247 

248 def __call__(self, tokens): 

249 primary_boost = self.primary_boost 

250 secondary_boost = self.secondary_boost 

251 combine = self.combine 

252 

253 for t in tokens: 

254 if combine: 

255 yield t 

256 

257 primary, secondary = double_metaphone(t.text) 

258 b = t.boost 

259 # Overwrite the token's text and boost and yield it 

260 if primary: 

261 t.text = primary 

262 t.boost = b * primary_boost 

263 yield t 

264 if secondary: 

265 t.text = secondary 

266 t.boost = b * secondary_boost 

267 yield t
